import json
import os
import shutil
import copy
import tkinter as tk
from tkinter import filedialog, messagebox
from tkinter import ttk
from PIL import Image, ImageTk
import webbrowser

products = []
filtered_products = []
file_path = ""
selected_index = None
current_sort_col = None
sort_reverse = False
dirty = False
undo_stack = []

# ---------------- HELPERS ----------------

def current_term():
    return (search_entry.get() or "").lower()

def rebuild_filtered():
    """Uvijek ponovo napravi filtered listu prema pretrazi + sort."""
    global filtered_products
    term = current_term()

    filtered_products = [
        p for p in products
        if term in str(p.get("sifra", "")).lower()
        or term in str(p.get("naziv", "")).lower()
    ]

    # primijeni sort ako postoji
    if current_sort_col:
        filtered_products.sort(
            key=lambda x: str(x.get(current_sort_col, "")).lower(),
            reverse=sort_reverse
        )

def clear_form():
    global selected_index
    selected_index = None
    for key in fields:
        if key in entries:
            entries[key].delete(0, tk.END)
    grupa_var.set("")
    active_var.set(True)
    image_label.config(image="")
    image_label.image = None

def mark_dirty():
    global dirty
    dirty = True

def push_undo():
    undo_stack.append(copy.deepcopy(products))

def undo():
    global products
    if not undo_stack:
        return
    products = undo_stack.pop()
    mark_dirty()
    rebuild_filtered()
    refresh_table()
    clear_form()

# ---------------- THEME ----------------

def set_dark():
    style.theme_use("clam")
    style.configure("Treeview",
                    background="#2b2b2b",
                    foreground="white",
                    fieldbackground="#2b2b2b")
    style.configure("Treeview.Heading",
                    background="#1e1e1e",
                    foreground="white")

def set_light():
    style.theme_use("clam")
    style.configure("Treeview",
                    background="white",
                    foreground="black",
                    fieldbackground="white")
    style.configure("Treeview.Heading",
                    background="#dddddd",
                    foreground="black")

# ---------------- LOAD / SAVE ----------------

def load_file():
    global products, file_path, dirty, current_sort_col, sort_reverse
    file_path = filedialog.askopenfilename(filetypes=[("JSON files", "*.json")])
    if not file_path:
        return

    with open(file_path, "r", encoding="utf-8") as f:
        products = json.load(f)

    dirty = False
    current_sort_col = None
    sort_reverse = False

    populate_groups()
    rebuild_filtered()
    refresh_table()
    clear_form()

def save_file():
    global dirty
    if not file_path:
        messagebox.showerror("Greška", "Nije učitan fajl")
        return

    with open(file_path, "w", encoding="utf-8") as f:
        json.dump(products, f, indent=2, ensure_ascii=False)

    dirty = False
    messagebox.showinfo("OK", "Sačuvano!")

# ---------------- PREVIEW (stabilno) ----------------

def open_preview():
    """Otvori index.html u pravom browseru (Edge/Chrome) – radi 100%."""
    if not file_path:
        messagebox.showerror("Greška", "Prvo učitaj JSON")
        return

    katalog_index = os.path.join(os.path.dirname(file_path), "..", "index.html")
    katalog_index = os.path.abspath(katalog_index)

    if not os.path.exists(katalog_index):
        messagebox.showerror("Greška", f"index.html nije pronađen:\n{katalog_index}")
        return

    webbrowser.open("file:///" + katalog_index.replace("\\", "/"))

# ---------------- SEARCH ----------------

def search():
    rebuild_filtered()
    refresh_table()
    clear_form()

# ---------------- SORT ----------------

def sort_by(column):
    global sort_reverse, current_sort_col
    if current_sort_col == column:
        sort_reverse = not sort_reverse
    else:
        current_sort_col = column
        sort_reverse = False

    rebuild_filtered()
    refresh_table()

# ---------------- TABLE ----------------

def refresh_table():
    tree.delete(*tree.get_children())
    for i, p in enumerate(filtered_products):
        tree.insert("", "end", iid=str(i),
                    values=(p.get("sifra"), p.get("naziv")))

def on_select(event):
    global selected_index
    sel = tree.selection()
    if not sel:
        return

    idx = int(sel[0])
    p = filtered_products[idx]

    # nađi stvarni index u products
    try:
        selected_index = products.index(p)
    except ValueError:
        selected_index = None
        return

    # popuni formu
    for key in fields:
        if key == "aktivno":
            active_var.set(p.get("aktivno", "DA") != "NE")
        elif key == "grupa":
            grupa_var.set(p.get("grupa", ""))
        else:
            entries[key].delete(0, tk.END)
            entries[key].insert(0, str(p.get(key, "")))

    show_image(p.get("slika"))

# ---------------- UPDATE ----------------

def update_product():
    global selected_index
    if selected_index is None:
        return

    push_undo()
    mark_dirty()

    p = products[selected_index]
    for key in fields:
        if key == "aktivno":
            p["aktivno"] = "DA" if active_var.get() else "NE"
        elif key == "grupa":
            p["grupa"] = grupa_var.get()
        else:
            p[key] = entries[key].get()

    populate_groups()
    rebuild_filtered()
    refresh_table()

# ---------------- NEW / DELETE ----------------

def new_product():
    push_undo()
    mark_dirty()

    empty = {k: "" for k in fields}
    empty["aktivno"] = "DA"
    empty["grupa"] = ""
    products.append(empty)

    populate_groups()
    rebuild_filtered()
    refresh_table()

    # selektuj novododani (zadnji u filtered možda nije zadnji zbog pretrage; zato očisti pretragu)
    search_entry.delete(0, tk.END)
    rebuild_filtered()
    refresh_table()
    tree.selection_set(str(len(filtered_products) - 1))
    on_select(None)

def delete_product():
    global selected_index
    if selected_index is None:
        return

    if not messagebox.askyesno("Potvrda", "Da li želite obrisati odabrani artikal?"):
        return

    push_undo()
    mark_dirty()

    products.pop(selected_index)
    selected_index = None

    populate_groups()
    rebuild_filtered()
    refresh_table()
    clear_form()

# ---------------- GROUPS ----------------

def populate_groups():
    groups = sorted(set(p.get("grupa", "") for p in products if p.get("grupa")))
    grupa_dropdown["values"] = groups
    grupa_dropdown["state"] = "normal"  # možeš upisati novu grupu ručno

# ---------------- IMAGE ----------------

def upload_image():
    if selected_index is None:
        return

    img_path = filedialog.askopenfilename(
        filetypes=[("Image files", "*.jpg *.png *.jpeg")]
    )
    if not img_path:
        return

    if not file_path:
        messagebox.showerror("Greška", "Prvo učitaj JSON")
        return

    images_folder = os.path.join(os.path.dirname(file_path), "..", "images")
    os.makedirs(images_folder, exist_ok=True)

    filename = os.path.basename(img_path)
    dest_path = os.path.join(images_folder, filename)
    shutil.copy(img_path, dest_path)

    push_undo()
    mark_dirty()

    entries["slika"].delete(0, tk.END)
    entries["slika"].insert(0, filename)
    products[selected_index]["slika"] = filename

    show_image(filename)

def show_image(filename):
    if not file_path or not filename:
        image_label.config(image="")
        image_label.image = None
        return

    img_path = os.path.join(os.path.dirname(file_path), "..", "images", filename)
    if not os.path.exists(img_path):
        image_label.config(image="")
        image_label.image = None
        return

    img = Image.open(img_path)
    img.thumbnail((150, 150))
    photo = ImageTk.PhotoImage(img)
    image_label.config(image=photo)
    image_label.image = photo

def on_closing():

/* GRID */
function setCols(n){
  grid.style.gridTemplateColumns = `repeat(${n}, minmax(0,1fr))`;
  localStorage.setItem("cols", n);
}
setCols(localStorage.getItem("cols") || "3");
colsSel.value = localStorage.getItem("cols") || "3";
colsSel.onchange = () => setCols(colsSel.value);

/* BADGE */
function renderBadge(x){
  if(!x.oznaka) return "";

  const o = x.oznaka.toUpperCase();

  if(o==="AKCIJA")
    return `<div class="badge badge-akcija">AKCIJA ${x.akcija_postotak ? "- "+x.akcija_postotak+"%" : ""}</div>`;

  if(o==="NOVO")
    return `<div class="badge badge-novo">NOVO</div>`;

  if(o==="1+1")
    return `<div class="badge badge-11">1+1 GRATIS</div>`;

  if(o==="ISTEK")
    return `<div class="badge badge-istek">PRI ISTEKU</div>`;

  if(o==="STIZE")
    return `<div class="badge badge-stize">STIŽE USKORO</div>`;

  return "";
}

/* KLASA KARTICE */
function cardClass(x){
  if(!x.oznaka) return "card";
  if(x.oznaka.toUpperCase()==="AKCIJA") return "card card-akcija";
  if(x.oznaka.toUpperCase()==="1+1") return "card card-11";
  return "card";
}

/* UČITAJ JOŠ ARTIKALA (LAZY RENDER) */
function ucitajJos(){

  let kraj = Math.min(prikazano + KORAK, filtered.length);
  let html = "";

  for(let i = prikazano; i < kraj; i++){
    const x = filtered[i];

    html += `
      <div class="${cardClass(x)}">
        ${renderBadge(x)}
        <img class="img" loading="lazy" decoding="async" src="images/${x.slika}" onerror="this.src='no-image.png'">
        <div class="t">${x.naziv}</div>
        <div class="meta">
          Šifra: <b>${x.sifra}</b><br>
          VPC: <b>${x.vpc}</b> KM | MPC: <b>${x.mpc}</b> KM<br> 
          Pakovanje: <b>${x.pakovanje}</b>
        </div>
      </div>
    `;
  }

  grid.insertAdjacentHTML("beforeend", html);
  prikazano = kraj;
}

/* RENDER */
function render(){

  const term = q.value.toLowerCase();
  const g = groupSel.value;

  filtered = items
    .filter(x => x && x.sifra && x.naziv)
    .filter(x => (x.aktivno||"").toUpperCase()!=="NE")
    .filter(x => !g || x.grupa===g)
    .filter(x => !term || (`${x.naziv} ${x.sifra}`).toLowerCase().includes(term))
    .sort((a,b)=>Number(a.redoslijed||0)-Number(b.redoslijed||0));

  grid.innerHTML = "";
  prikazano = 0;
  ucitajJos();
}

/* SKROL DETEKCIJA */
window.addEventListener("scroll", () => {
  if(window.innerHeight + window.scrollY >= document.body.offsetHeight - 500){
    ucitajJos();
  }
});

/* GRUPE */
function fillGroups(){
  const groups=[...new Set(items.map(x=>x.grupa).filter(Boolean))];
  groupSel.innerHTML=`<option value="">Sve grupe</option>`+
    groups.map(g=>`<option>${g}</option>`).join("");
}

q.oninput=render;
groupSel.onchange=render;

      slika: r.slika,
      oznaka: r.oznaka,
      akcija_postotak: r.akcija_postotak,
      aktivno: r.aktivno
    }));

    fillGroups();
    render();
  })
  .catch(err => {
    console.error("Greška pri učitavanju JSON-a:", err);
  });

# ---------------- HELPERS ----------------

def current_term():
    return (search_entry.get() or "").lower()

def rebuild_filtered():
    global filtered_products

    term = current_term()
    status = status_var.get() if 'status_var' in globals() else "Svi"
    grupa_filter = group_filter_var.get() if 'group_filter_var' in globals() else "Sve"

    filtered_products = []

    for p in products:

        # SEARCH
        if term not in str(p.get("sifra", "")).lower() and \
           term not in str(p.get("naziv", "")).lower():
            continue

        # STATUS
        if status == "Aktivni" and p.get("aktivno") == "NE":
            continue

        if status == "Neaktivni" and p.get("aktivno") != "NE":
            continue

        # GRUPA
        if grupa_filter != "Sve" and p.get("grupa") != grupa_filter:
            continue

        filtered_products.append(p)

    if current_sort_col:
        filtered_products.sort(
            key=lambda x: str(x.get(current_sort_col, "")).lower(),
            reverse=sort_reverse
        )

def clear_form():
    global selected_index
    selected_index = None
    for key in fields:
        if key in entries:
            entries[key].delete(0, tk.END)
    grupa_var.set("")
    active_var.set(True)
    image_label.config(image="")
    image_label.image = None

def mark_dirty():
    global dirty
    dirty = True

def push_undo():
    undo_stack.append(copy.deepcopy(products))

def undo():
    global products
    if not undo_stack:
        return
    products = undo_stack.pop()
    mark_dirty()
    rebuild_filtered()
    refresh_table()
    clear_form()

def sort_by(column):
    global sort_reverse, current_view_mode

    current_view_mode = "sorted"
    sort_reverse = not sort_reverse

    items = list(tree.get_children())

    def get_value(iid):
        values = tree.item(iid)["values"]
        if column == "sifra":
            return str(values[1]).lower()
        if column == "naziv":
            return str(values[2]).lower()
        return ""

    items.sort(key=get_value, reverse=sort_reverse)

    for index, iid in enumerate(items):
        tree.move(iid, "", index)

def sort_by_redoslijed():

    refresh_table()

            values=(

        p["redoslijed"] = i + 1

def on_tree_press(event):
    global drag_item
    item = tree.identify_row(event.y)
    if item:
        drag_item = item
        tree.selection_set(item)
        tree.item(item, tags=("dragging",))

def on_tree_motion(event):
    global drag_item
    if not drag_item:
        return

    target = tree.identify_row(event.y)
    if not target or target == drag_item:
        return

    tree.move(drag_item, "", tree.index(target))

def on_tree_release(event):
    global drag_item, current_view_mode

    if not drag_item:
        return

    # ako nisi u originalnom (#) modu – samo prekini tiho
    if current_view_mode != "order": 
        drag_item = None
        return

    push_undo()
    mark_dirty()

    # redoslijed samo za trenutno prikazane (filtered) u tabeli
    ordered_uids = list(tree.get_children())
    uid_to_p = {p["uid"]: p for p in products}

    # napravi listu proizvoda u tom novom redoslijedu (samo filtered subset)
    ordered_subset = [uid_to_p[uid] for uid in ordered_uids if uid in uid_to_p]
    subset_uids = set(p["uid"] for p in ordered_subset)
    it = iter(ordered_subset)

    # sastavi products tako da NE-dirnuti ostanu na mjestu,
    # a samo filtered dio dobije novi poredak
    new_products = []
    for p in products:
        if p.get("uid") in subset_uids:
            new_products.append(next(it))
        else:
            new_products.append(p)

    products[:] = new_products

    update_redoslijed()
    rebuild_filtered()
    refresh_table()

    tree.item(drag_item, tags=())
    drag_item = None

    for key in fields:
        if key == "aktivno":
            active_var.set(p.get("aktivno", "DA") != "NE")

            grupa_var.set(p.get("grupa", ""))
        else:
            entries[key].delete(0, tk.END)
            entries[key].insert(0, str(p.get(key, "")))

    show_image(p.get("slika"))
